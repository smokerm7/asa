# sum_analysis.py

import timeit
import matplotlib.pyplot as plt
import random

# Базовая простая функция

def compute_sum():
    """Вычисляет сумму двух чисел, введенных пользователем."""
    x = int(input())  # O(1) - считывание строки и конвертация
    y = int(input())  # O(1)
    sum_result = x + y    # O(1) - математическая операция
    print(sum_result)     # O(1) - вывод результата
    # Общая асимптотика: O(1)

# compute_sum()  # Раскомментировать для тестирования базовой функции

# УСЛОЖНЕННАЯ ВЕРСИЯ ДЛЯ ИССЛЕДОВАНИЯ ПРОИЗВОДИТЕЛЬНОСТИ
 # Сложение N чисел для иллюстрации линейной асимптотики O(N)

def calculate_array_sum(numbers):
    """Вычисляет сумму всех элементов в списке.
    Асимптотика: O(N), где N - количество элементов.
    """
    sum_total = 0
    # O(1) - объявление переменной
    for element in numbers:           # O(N) - обход всех элементов списка
        sum_total += element      # O(1) - сложение и присваивание
    return sum_total              # O(1) - возврат значения
    # Итоговая сложность: O(1) + O(N) * O(1) + O(1) = O(N)

# Функция для хронометража выполнения

def time_execution(function, input_data):
    """Замеряет время работы функции в миллисекундах."""
    timer_start = timeit.default_timer()
    function(input_data)
    timer_end = timeit.default_timer()
    return (timer_end - timer_start) * 1000  # Перевод в мс


# Спецификации компьютера (указать актуальные данные)
system_specs = """
Спецификации тестовой системы:
- Процессор: Intel Core i5-12400
- Оперативная память: 32 GB
- Видеокарта: NVIDIA GeForce RTX 3070 Ti
- Операционная система: Windows 10 Pro
- Интерпретатор Python: 3.12
"""
print(system_specs)

# Проведение тестов производительности
dimensions = [1000, 5000, 10000, 50000, 100000, 500000]  # Размерности массивов
timings = []  # Замеры времени для каждой размерности

print("Результаты хронометража для функции суммирования массива:")
print("{:>10} {:>12} {:>15}".format(
    "Размер (N)", "Время (мс)", "Время/N (мкс)"))

for dim in dimensions:  # Создание массива случайных чисел заданной длины

    # Хронометраж выполнения (усреднение по 10 запускам)
    test_data = [random.randint(1, 1000) for _ in range(dim)]

    avg_time = timeit.timeit(
        lambda: calculate_array_sum(test_data), number=10) * 1000 / 10

    timings.append(avg_time)
    element_time = (avg_time * 1000) / \
        dim if dim > 0 else 0  # микросекунды на элемент

    print("{:>10} {:>12.4f} {:>15.4f}".format(
        dim, avg_time, element_time))

# Построение графика
plt.figure(figsize=(10, 6))
plt.plot(dimensions, timings, 'ro-', label='Экспериментальные замеры')
plt.xlabel('Количество элементов (N)')
plt.ylabel('Длительность выполнения (мс)')
plt.title('Зависимость времени работы от размера массива\nАсимптотика: O(N)')
plt.grid(True, which='both', linestyle='--', linewidth=0.5)
plt.legend()
plt.savefig('./report/complexity_graph.png', dpi=300, bbox_inches='tight')
plt.show()

# Дополнительная аналитика: сопоставление с теоретической моделью
print("\nИнтерпретация результатов:")
print("1. Теоретическая оценка сложности: O(N)")
print("2. Экспериментальные данные подтверждают линейную зависимость")
print("3. Среднее время обработки элемента: ~{:.4f} мкс".format(
    element_time))
